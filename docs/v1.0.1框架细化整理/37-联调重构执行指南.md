```typescript
async function getCharacter(id: number): Promise<Character> {
  const { data } = await client.get(`/character/${id}`)
  return data as Character
}

async function updateCharacter(id: number, payload: Partial<CharacterCreateRequest>): Promise<Character> {
  const { data } = await client.put(`/character/${id}`, payload)
  return data as Character
}

async function uploadAvatar(id: number, file: File): Promise<{ avatar: string }> {
  const formData = new FormData()
  formData.append('file', file)
  const { data } = await client.post(`/character/${id}/avatar`, formData, {
    headers: { 'Content-Type': 'multipart/form-data' },
  })
  return data as { avatar: string }
}

async function getCharacterStats() {
  const { data } = await client.get('/character/stats')
  return data
}

async function levelupCharacter(id: number, experiencePoints: number) {
  const { data } = await client.post(`/character/${id}/levelup`, { experience_points: experiencePoints })
  return data
}
```
